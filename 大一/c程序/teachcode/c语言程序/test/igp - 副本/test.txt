struct shuju* creat();


struct shuju{
	int s;
	struct shuju *next;
};
struct shuju* creat(){
	int shu;
	int size;
	size=sizeof(struct shuju);
	struct shuju *head,*tail,*p;
	
	head=tail=NULL;
	FILE *fp;
	fp=fopen("a.txt","r+");
	fscanf(fp,"%d",&shu);
	printf("%d",shu);
	while(shu){
		p=(struct shuju*)malloc(size);
		p->s=shu;
		p->next=NULL;
		if(head==NULL)
		head=p;
		else
		tail->next=p;
		tail=p;
		if(fscanf(fp,"%d",&shu)==-1)break;
		printf("%d",shu);
	}
	fclose(fp);
	return head;
};
void Main()
{
	int geshu=0;
	struct shuju*p,*head;
	p=creat();
	head=p;
	while(p){
		geshu++;
		p=p->next;
	}
	SetWindowTitle("画些什么东西");
	InitGraphics();
	MovePen(0.5,0.5);
	DrawLine(geshu,0);
}
