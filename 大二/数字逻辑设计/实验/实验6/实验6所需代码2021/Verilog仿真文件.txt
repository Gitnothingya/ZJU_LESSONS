`timescale 1ns / 1ps

module MC14495Test();

	// Inputs
	reg [3:0] D;
	reg LE;
	reg Point;

	// Outputs
	wire [6:0] LED;
	wire p;

	// Instantiate the Unit Under Test (UUT)
	MyMC14495 uut (
		.D(D), 
		.LE(LE), 
		.Point(Point), 
		.LED(LED), 
		.p(p)
	);

	integer i;
	initial begin
		D = 4'b0000;
		LE = 1;
		Point = 0;
		
		for (i=0; i<=15;i=i+1) begin
			D = i;
			Point = i;
		#50;
		end
		
		#50;
		LE = 0;
	end
      
endmodule
